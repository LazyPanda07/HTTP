280 string HTTPBuilder::build(const string& data, const unordered_map<string, string>& additionalHeaders) const
281 {
282 string result;
283 unordered_map<string, string> buildHeaders(additionalHeaders);
284
285 if (data.size())
286 {
287 buildHeaders["Content-Length"] = to_string(data.size());
288 }
289 else if (_chunks.size())
290 {
291 buildHeaders["Transfer-Encoding"] = "chunked";
292 }
293
294 if (method.empty()) // response
295 {
296 result = string(_HTTPVersion) + " " + _responseCode + "\r\n" + _headers;
297 }
298 else // request
299 {
300 result = method + ' ';
301
302 if (_parameters.empty())
303 {
304 result += "/";
305 }
306
307 result += _parameters + ' ' + _HTTPVersion + "\r\n" + _headers;
308 }
309
310 for (const auto& [header, value] : buildHeaders)
311 {
312 result += header + ": " + value + "\r\n";
313 }
314
315 result += "\r\n";
316
317 if (data.size())
318 {
319 result += data;
320 }
321 else if (_chunks.size())
322 {
323 result += HTTPBuilder::getChunks(_chunks);
324 }
325
326 return result;
327 }
