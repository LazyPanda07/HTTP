106 string HTTPBuilder::getChunks(const vector<string>& chunks, bool partialChunks, bool preCalculateSize)
107 {
108 string result;
109
110 if (preCalculateSize)
111 {
112 size_t resultSize = 0;
113
114 for (const string& chunk : chunks)
115 {
116 resultSize += format("{:x}", chunk.size()).size() + HTTPParser::crlf.size() + chunk.size() + HTTPParser::crlf.size();
117 }
118
119 resultSize += 1 + HTTPParser::crlf.size();
120
121 result.reserve(resultSize);
122 }
123
124 for (const string& chunk : chunks)
125 {
126 result += HTTPBuilder::getChunk(chunk);
127 }
128
129 if (!partialChunks)
130 {
131 result += HTTPBuilder::getChunk({});
132 }
133
134 return result;
135 }

293 string HTTPBuilder::build(string_view data, const unordered_map<string, string>& additionalHeaders) const
294 {
295 string result;
296 unordered_map<string, string> buildHeaders(additionalHeaders);
297
298 if (data.size())
299 {
300 buildHeaders["Content-Length"] = to_string(data.size());
301 }
302 else if (_chunks.size())
303 {
304 buildHeaders["Transfer-Encoding"] = "chunked";
305 }
306
307 if (method.empty())
308 {
309 result = format("{} {}{}{}", _HTTPVersion, _responseCode, HTTPParser::crlf, _headers);
310 }
311 else
312 {
313 result = method + ' ';
314
315 if (_parameters.empty() && method != "CONNECT")
316 {
317 result += "/";
318 }
319
320 result += format("{} {}{}{}", _parameters, _HTTPVersion, HTTPParser::crlf, _headers);
321 }
322
323 for (const auto& [header, value] : buildHeaders)
324 {
325 result += format("{}: {}{}", header, value, HTTPParser::crlf);
326 }
327
328 result += HTTPParser::crlf;
329
330 if (data.size())
331 {
332 result += data;
333 }
334 else if (_chunks.size())
335 {
336 result += HTTPBuilder::getChunks(_chunks, _partialChunks);
337 }
338
339 return result;
340 }
